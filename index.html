function switchView(viewId) {

// Hide all layout containers

document.querySelectorAll('.view-panel').forEach(view => view.classList.add('hidden'));

// Un-hide targeted view

document.getElementById('view-' + viewId).classList.remove('hidden');

// Toggle highlight styles on active navigation menu buttons

document.querySelectorAll('.nav-btn').forEach(btn => {

btn.classList.remove('bg-indigo-600', 'text-white');

btn.classList.add('hover:bg-slate-800', 'text-slate-300');

});

event.currentTarget.classList.add('bg-indigo-600', 'text-white');

event.currentTarget.classList.remove('hover:bg-slate-800', 'text-slate-300');

}

function toggleIdRules() {

    const idType = document.getElementById('idType').value;

    const expiryInput = document.getElementById('idExpiry');

    const alertBox = document.getElementById('compliance-alert');

    

    // Clear out active warnings

    alertBox.classList.add('hidden');

    

    if (idType === 'Credible Witness') {

        alertBox.textContent = "⚖️ Statutory Check: Credible witnesses must be personally known to the notary or present their own valid credentials under separate entries.";

        alertBox.classList.remove('hidden');

        return;

    }


    if (expiryInput.value) {

        const expiryDate = new Date(expiryInput.value);

        const today = new Date();

        

        if (expiryDate <= today) {

            alertBox.textContent = "🚨 Non-Compliant: Stored Identification is expired. State regulations prohibit proceeding with expired identification documents.";

            alertBox.classList.remove('hidden');

        }

    }

}


// Track input dates for state validation triggers

document.getElementById('idExpiry').addEventListener('change', toggleIdRules);
